def Trimestre(mes):
    ''' 
      Calcula el trimestre correspondiente al ao
      
      Parmetro: 
          mes: Mes del que queremos saber el trimestre
      Resultado:
          Valor del trimestre en funcin del mes recibido como parmetro
    '''
    if mes >= 1 and mes <= 3:
        return 1
    elif mes >= 4 and mes <= 6:
        return 2
    elif mes >= 7 and mes <= 9:
        return 3
    elif mes >= 10 and mes <= 12:
        return 4
